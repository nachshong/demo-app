# initial new repository 'demo-rep'
mkdir demo-rep
cd demo-rep
git init

# list of branches
git branch
- or - (show hash and subject)
git branch -v
- or - (show upstream branch also)
git branch -vv
- or - (show remote-tracking also)
git branch -a

# list of remotes
git remote
- or -
git remote -v

# add remote origin
git remote add origin <URL>

# override local repository
git fetch 
git reset --hard origin/master

#set upstream
git branch -u origin/master

# fetch all last changes from remote
git pull
- or -
git fetch
git merge

# push all last changes to remote
git add .
git commit
git push

# fix last commit author after push
git commit --amend --reset-author
git push --force
